        // Dropdown functionality for navigation

        document.addEventListener('DOMContentLoaded', function() {

            const dropdownGroups = document.querySelectorAll('.group.relative');


            dropdownGroups.forEach(group => {

                const trigger = group.querySelector('a'); // The main nav link that triggers the dropdown

                const menu = group.querySelector('.dropdown-menu');


                if (trigger && menu) {

                    // Show dropdown on hover

                    group.addEventListener('mouseenter', () => {

                        menu.classList.add('show');

                    });


                    // Hide dropdown when mouse leaves the group

                    group.addEventListener('mouseleave', () => {

                        menu.classList.remove('show');

                    });


                    // Handle keyboard navigation for accessibility

                    let timeoutId;

                    trigger.addEventListener('focus', () => {

                        clearTimeout(timeoutId); // Clear any pending hide

                        menu.classList.add('show');

                    });


                    // Hide when focus leaves the dropdown or its trigger

                    group.addEventListener('focusout', (event) => {

                        // Check if the new focused element is outside the dropdown group

                        if (!group.contains(event.relatedTarget)) {

                            // Delay hiding to allow focus to move within the dropdown

                            timeoutId = setTimeout(() => {

                                menu.classList.remove('show');

                            }, 100); // Small delay to allow for quick mouse movements

                        }

                    });

                }
